Function useCreateTables

Custom hook to create tables in the database. This hook generates SQL statements to create tables based on the provided schema and executes them in a transaction. It provides state and callbacks for handling the creation process.

Example

import { useCreateTables } from 'reactite'

const App = ()=>{

 const { retry, creating, error, sql, status, success, tables } = useCreateTables(
   {
      posts: {
        fields: {
          title: field.unique().type("TEXT"),
        },
        timestamps: true,
       snakeCase: true,
       },
     users: {
       fields: {
          username: field.unique().type("TEXT"),
         password: field.type("TEXT"),
         avatar: field.type("TEXT").nullable().default("hello.jpg"),
         id: field.type("INTEGER").pk().autoIncrement(),
       },
       snakeCase: false,
       timestamps: false,
       skipIfExist: true,
     },
    },
   { skipIfExist: true }
 );
return null
)
